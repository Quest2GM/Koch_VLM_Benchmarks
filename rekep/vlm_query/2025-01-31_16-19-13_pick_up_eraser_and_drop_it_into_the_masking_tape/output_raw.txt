# There are two stages involved in this task:
# 1. Grasp the eraser.
# 2. Drop the eraser into the masking tape.

    """Align the end-effector with the eraser (keypoint 10)."""
    eraser_position = keypoints[10]
    cost = np.linalg.norm(end_effector - eraser_position)

    offsetted_point = masking_tape_position + np.array([0, 0, 10])  # 10 cm above
    cost = np.linalg.norm(keypoints[10] - offsetted_point)
    return cost
